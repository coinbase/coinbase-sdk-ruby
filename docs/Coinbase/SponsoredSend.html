def sign(key)
  raise unless key.is_a?(Eth::Key)
  raise Coinbase::AlreadySignedError if signed?

  @signature = Eth::Util.prefix_hex(key.sign(Eth::Util.hex_to_bin(typed_data_hash)))
end
